//Currently, this is used to wait for a new page to load
    def waitForPageLoad() {

        def js = getJs()
        //def status = js.exec("return document.readyState")
        waitFor() {js.exec("return document.readyState") == "complete"}

    }

    //Currently, this is used to wait for ajax calls to finish
    def waitForAjaxComplete(waitProfile = null) {

        def js = getJs()

        if(!waitProfile) {
            waitFor() { !js.exec("return jQuery.active") }
        } else {
            waitFor(waitProfile) { !js.exec("return jQuery.active") }
        }

    }
